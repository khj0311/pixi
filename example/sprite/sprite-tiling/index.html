PIXI Sprite - Tiling Sprite
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.spriteTiling = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-sprite-tiling-sprite');

            if (!!obj.section) {
                setElements();
                setTicker();
            }
        };

        var setElements = function() {
            obj.count = 0;
            obj.app = new PIXI.Application({
                view: document.getElementById('spriteTiling'),
                backgroundColor: 0x1099bb
            });

            obj.texture = PIXI.Texture.from('/content/images/tile.jpg');
            obj.tilingSprite = new PIXI.TilingSprite(
                obj.texture,
                obj.app.screen.width,
                obj.app.screen.height
            );

            obj.app.stage.addChild(obj.tilingSprite);
        };

        var setTicker = function() {
            obj.app.ticker.add(function(delta) {
                obj.count += 0.005;

                obj.tilingSprite.tileScale.x = 2 + Math.sin(obj.count);
                obj.tilingSprite.tileScale.y = 2 + Math.cos(obj.count);

                obj.tilingSprite.tilePosition.x += 1;
                obj.tilingSprite.tilePosition.y += 1;
            });
        };

        return {
            init: init
        }
    })();

    pixi.component.spriteTiling.init();
})();

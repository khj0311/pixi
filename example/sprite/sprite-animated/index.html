PIXI Sprite - Animated Sprite - Explosion
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.spriteAnimated = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-sprite-animated-explosion');

            if (!!obj.section) {
                setElements();
                setLoader();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('spriteAnimated')
            });
            obj.app.stop();
        };

        var setLoader = function() {
            obj.app.loader
                .add('spritesheet', '/content/assets/explosion.json')
                .load(onAssetsLoadded);
        };

        var onAssetsLoadded = function() {
            var explosionTextures = [];
            var i = 0;

            for (i = 0; i < 26; i++) {
                var texture = PIXI.Texture.from(`Explosion_Sequence_A_${i + 1}.png`);
                explosionTextures.push(texture);
            }

            for (i = 0; i < 50; i++) {
                var explosion = new PIXI.AnimatedSprite(explosionTextures);

                explosion.x = Math.random() * obj.app.screen.width;
                explosion.y = Math.random() * obj.app.screen.height;
                explosion.anchor.set(0.5);
                explosion.rotation = Math.random() * Math.PI;
                explosion.scale.set(0.75 + Math.random() * 0.5);
                explosion.gotoAndPlay(Math.random() * 27);
                obj.app.stage.addChild(explosion);
            }

            obj.app.start();
        };

        return {
            init: init
        }
    })();

    pixi.component.spriteAnimated.init();
})();

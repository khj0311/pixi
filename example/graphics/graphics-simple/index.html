PIXI Graphics - Simple
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.graphicsSimple = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-graphics-simple');

            if (!!obj.section) {
                setElements();
                drawGraphics();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('graphicsSimple'),
                antialias: true
            });

            obj.graphics = new PIXI.Graphics();
        };

        var drawGraphics = function() {
            obj.graphics.beginFill(0xDE3249);
            obj.graphics.drawRect(50, 50, 100, 100);
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0xFEEB77, 1);
            obj.graphics.beginFill(0x650A5A);
            obj.graphics.drawRect(200, 50, 100, 100);
            obj.graphics.endFill();

            obj.graphics.lineStyle(10, 0xFFBD01, 1);
            obj.graphics.beginFill(0xC34288);
            obj.graphics.drawRect(350, 50, 100, 100);
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0xFFFFFF, 1);
            obj.graphics.beginFill(0xAA4F08);
            obj.graphics.drawRect(530, 50, 140, 100);
            obj.graphics.endFill();

            obj.graphics.lineStyle(0);
            obj.graphics.beginFill(0xDE3249, 1);
            obj.graphics.drawCircle(100, 250, 50);
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0xFEEB77, 1);
            obj.graphics.beginFill(0x650A5A, 1);
            obj.graphics.drawCircle(250, 250, 50);
            obj.graphics.endFill();

            obj.graphics.lineStyle(10, 0xFFBD01, 1);
            obj.graphics.beginFill(0xC34288, 1);
            obj.graphics.drawCircle(400, 250, 50);
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0xFFFFFF, 1);
            obj.graphics.beginFill(0xAA4F08, 1);
            obj.graphics.drawEllipse(600, 250, 80, 50);
            obj.graphics.endFill();

            obj.graphics.beginFill(0xFF3300);
            obj.graphics.lineStyle(4, 0xffd900, 1);
            obj.graphics.moveTo(50, 350);
            obj.graphics.lineTo(250, 350);
            obj.graphics.lineTo(100, 400);
            obj.graphics.lineTo(50, 350);
            obj.graphics.closePath();
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0xFF00FF, 1);
            obj.graphics.beginFill(0x650A5A, 0.25);
            obj.graphics.drawRoundedRect(50, 440, 100, 100, 16);
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0xFFFFFF);
            obj.graphics.beginFill(0x35CC5A, 1);
            obj.graphics.drawStar(360, 370, 5, 50);
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0xFFFFFF);
            obj.graphics.beginFill(0xFFCC5A, 1);
            obj.graphics.drawStar(280, 510, 7, 50);
            obj.graphics.endFill();

            obj.graphics.lineStyle(4, 0xFFFFFF);
            obj.graphics.beginFill(0x55335A, 1);
            obj.graphics.drawStar(470, 450, 4, 50);
            obj.graphics.endFill();

            const path = [600, 370, 700, 460, 780, 420, 730, 570, 590, 520];

            obj.graphics.lineStyle(0);
            obj.graphics.beginFill(0x3500FA, 1);
            obj.graphics.drawPolygon(path);
            obj.graphics.endFill();

            obj.app.stage.addChild(obj.graphics);
        };

        return {
            init: init
        }
    })();

    pixi.component.graphicsSimple.init();
})();

PIXI Graphics - Advanced
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.graphicsAdvanced = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-graphics-advanced');

            if (!!obj.section) {
                setElements();
                drawRealPath();
                drawBezier();
                drawRealPath2();
                drawBezier2();
                drawArcs();
                drawHole();
                drawBeatifulRect();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('graphicsAdvanced'),
                antialias: true
            });

            obj.sprite = PIXI.Sprite.from('/content/images/bg_rotate.jpg');
        };

        var drawRealPath = function() {
            var realPath = new PIXI.Graphics();

            realPath.lineStyle(2, 0xFFFFFF, 1);
            realPath.moveTo(0, 0);
            realPath.lineTo(100, 200);
            realPath.lineTo(200, 200);
            realPath.lineTo(240, 100);

            realPath.position.x = 50;
            realPath.position.y = 50;

            obj.app.stage.addChild(realPath);
        };

        var drawBezier = function() {
            var bezier = new PIXI.Graphics();

            bezier.lineStyle(5, 0xAA0000, 1);
            bezier.bezierCurveTo(100, 200, 200, 200, 240, 100);

            bezier.position.x = 50;
            bezier.position.y = 50;

            obj.app.stage.addChild(bezier);
        };

        var drawRealPath2 = function() {
            var realPath2 = new PIXI.Graphics();

            realPath2.lineStyle(2, 0xFFFFFF, 1);
            realPath2.moveTo(0, 0);
            realPath2.lineTo(0, -100);
            realPath2.lineTo(150, 150);
            realPath2.lineTo(240, 100);

            realPath2.position.x = 320;
            realPath2.position.y = 150;

            obj.app.stage.addChild(realPath2);
        };

        var drawBezier2 = function() {
            var bezier2 = new PIXI.Graphics();

            bezier2.lineTextureStyle({ width: 10, texture: obj.sprite.texture });
            bezier2.bezierCurveTo(0, -100, 150, 150, 240, 100);

            bezier2.position.x = 320;
            bezier2.position.y = 150;

            obj.app.stage.addChild(bezier2);
        };

        var drawArcs = function() {
            var arc = new PIXI.Graphics();

            arc.lineStyle(5, 0xAA00BB, 1);
            arc.arc(600, 100, 50, Math.PI, 2 * Math.PI);

            obj.app.stage.addChild(arc);

            var arc2 = new PIXI.Graphics();
            
            arc2.lineStyle(6, 0x3333DD, 1);
            arc2.arc(650, 270, 60, 2 * Math.PI, 3 * Math.PI / 2);
            
            obj.app.stage.addChild(arc2);

            var arc3 = new PIXI.Graphics();
            
            arc3.lineTextureStyle({ width: 20, texture: obj.sprite.texture });
            arc3.arc(650, 420, 60, 2 * Math.PI, 2.5 * Math.PI / 2);
            
            obj.app.stage.addChild(arc3);
        };

        var drawHole = function() {
            var rectAndHole = new PIXI.Graphics();

            rectAndHole.beginFill(0x00FF00);
            rectAndHole.drawRect(350, 350, 150, 150);
            rectAndHole.beginHole();
            rectAndHole.drawCircle(375, 375, 25);
            rectAndHole.drawCircle(425, 425, 25);
            rectAndHole.drawCircle(475, 475, 25);
            rectAndHole.endHole();
            rectAndHole.endFill();

            obj.app.stage.addChild(rectAndHole);
        };

        var drawBeatifulRect = function() {
            var beatifulRect = new PIXI.Graphics();

            beatifulRect.lineTextureStyle({ width: 20, texture: obj.sprite.texture });
            beatifulRect.beginFill(0xFF0000);
            beatifulRect.drawRect(80, 350, 150, 150);
            beatifulRect.endFill();

            obj.app.stage.addChild(beatifulRect);
        };

        return {
            init: init
        }
    })();

    pixi.component.graphicsAdvanced.init();
})();

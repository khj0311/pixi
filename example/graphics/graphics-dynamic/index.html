PIXI Graphics - Dynamic
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.graphicsDynamic = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-graphics-dynamic');

            if (!!obj.section) {
                setElements();
                drawGraphics();
                drawMovingShape();
                setTicker();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('graphicsDynamic'),
                antialias: true
            });
            obj.app.stage.interactive = true;

            obj.graphics = new PIXI.Graphics();
            obj.thing = new PIXI.Graphics();
            obj.count = 0;
        };

        var drawGraphics = function() {
            obj.graphics.beginFill(0xFF3300);
            obj.graphics.lineStyle(10, 0xffd900, 1);

            obj.graphics.moveTo(50, 50);
            obj.graphics.lineTo(250, 50);
            obj.graphics.lineTo(100, 100);
            obj.graphics.lineTo(250, 220);
            obj.graphics.lineTo(50, 220);
            obj.graphics.lineTo(50, 50);
            obj.graphics.closePath();
            obj.graphics.endFill();

            obj.graphics.lineStyle(10, 0xFF0000, 0.8);
            obj.graphics.beginFill(0xFF700B, 1);

            obj.graphics.moveTo(210, 300);
            obj.graphics.lineTo(450, 320);
            obj.graphics.lineTo(570, 350);
            obj.graphics.quadraticCurveTo(600, 0, 480, 100);
            obj.graphics.lineTo(330, 120);
            obj.graphics.lineTo(410, 200);
            obj.graphics.lineTo(210, 300);
            obj.graphics.closePath();
            obj.graphics.endFill();

            obj.graphics.lineStyle(2, 0x0000FF, 1);
            obj.graphics.drawRect(50, 250, 100, 100);

            obj.graphics.lineStyle(0);
            obj.graphics.beginFill(0xFFFF0B, 0.5);
            obj.graphics.drawCircle(470, 200, 100);
            obj.graphics.endFill();

            obj.graphics.lineStyle(20, 0x33FF00);
            obj.graphics.moveTo(30, 30);
            obj.graphics.lineTo(600, 300);


            obj.app.stage.addChild(obj.graphics);
        };

        var drawMovingShape = function() {
            obj.app.stage.addChild(obj.thing);
            obj.thing.x = 800 / 2;
            obj.thing.y = 600 / 2;

            window.app = obj.app;
            obj.app.renderer.plugins.interaction.on('pointerdown', onPointerDown);
        };

        var onPointerDown = function() {
            obj.graphics.lineStyle(Math.random() * 30, Math.random() * 0xFFFFFF, 1);
            obj.graphics.moveTo(Math.random() * 800, Math.random() * 600);
            obj.graphics.bezierCurveTo(
                Math.random() * 800, Math.random() * 600,
                Math.random() * 800, Math.random() * 600,
                Math.random() * 800, Math.random() * 600,
            );
        };

        var setTicker = function() {
            obj.app.ticker.add(function() {
                obj.count += 0.1;

                obj.thing.clear();
                obj.thing.lineStyle(10, 0xff0000, 1);
                obj.thing.beginFill(0xffFF00, 0.5);

                obj.thing.moveTo(-120 + Math.sin(obj.count) * 20, -100 + Math.cos(obj.count) * 20);
                obj.thing.lineTo(120 + Math.cos(obj.count) * 20, -100 + Math.sin(obj.count) * 20);
                obj.thing.lineTo(120 + Math.sin(obj.count) * 20, 100 + Math.cos(obj.count) * 20);
                obj.thing.lineTo(-120 + Math.cos(obj.count) * 20, 100 + Math.sin(obj.count) * 20);
                obj.thing.lineTo(-120 + Math.sin(obj.count) * 20, -100 + Math.cos(obj.count) * 20);
                obj.thing.closePath();

                obj.thing.rotation = obj.count * 0.1;
            });
        };

        return {
            init: init
        }
    })();

    pixi.component.graphicsDynamic.init();
})();

PIXI Interaction - Custom Hitarea
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.interactionCustomHitarea = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-interaction-custom-hitarea');

            if (!!obj.section) {
                setElements();
                setStarButton1();
                setStarButton2();
                setStarButton3();
                setStarButton4();
                setText();
                setStage();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('interactionCustomHitarea')
            });

            obj.yellowStar = PIXI.Texture.from('/content/images/yellowstar.png');
            obj.starButton1 = new PIXI.Sprite(obj.yellowStar);
            obj.starButton2 = new PIXI.Sprite(obj.yellowStar);
            obj.starButton3 = new PIXI.Sprite(obj.yellowStar);
            obj.starButton4 = new PIXI.Sprite(obj.yellowStar);
        };

        var setStarButton1 = function() {
            obj.starButton1.position.set(50, 200);

            obj.starButton1.buttonMode = true;
            obj.starButton1.interactive = true;

            obj.starButton1.on('pointerdown', function() {
                onClick(obj.starButton1);
            });
            obj.starButton1.on('pointerover', function() {
                onPointerOver(obj.starButton1);
            });
            obj.starButton1.on('pointerout', function() {
                onPointerOut(obj.starButton1);
            });
        };

        var setStarButton2 = function() {
            obj.starButton2.position.set(250, 200);
            obj.starButton2.hitArea = new PIXI.Polygon([
                80, 0,
                100, 50,
                160, 55,
                115, 95,
                130, 150,
                80, 120,
                30, 150,
                45, 95,
                0, 55,
                60, 50,
            ]);
            obj.starButton2.buttonMode = true;
            obj.starButton2.interactive = true;

            obj.starButton2.on('pointerdown', function() {
                onClick(obj.starButton2);
            });
            obj.starButton2.on('pointerover', function() {
                onPointerOver(obj.starButton2);
            });
            obj.starButton2.on('pointerout', function() {
                onPointerOut(obj.starButton2);
            });
        };

        var setStarButton3 = function() {
            obj.starButton3.position.set(450, 200);

            obj.starButton3.buttonMode = true;
            obj.starButton3.interactive = true;

            obj.squareMask = new PIXI.Graphics()
                .beginFill(0xFFFFFF)
                .drawRect(obj.starButton3.x, obj.starButton3.y, 75, 200)
                .endFill();

            obj.starButton3.mask = obj.squareMask;
            obj.starButton3.on('pointerdown', function() {
                onClick(obj.starButton3);
            });
            obj.starButton3.on('pointerover', function() {
                onPointerOver(obj.starButton3);
            });
            obj.starButton3.on('pointerout', function() {
                onPointerOut(obj.starButton3);
            });
        };

        var setStarButton4 = function() {
            obj.starButton4.position.set(600, 200);
            obj.squareMask2 = new PIXI.Graphics()
                .beginFill(0xFFFFFF)
                .drawRect(obj.starButton4.x, obj.starButton4.y, 75, 200)
                .endFill();

            obj.starButton4.mask = obj.squareMask2;
            obj.starButton4.hitArea = new PIXI.Polygon([
                80, 0,
                100, 50,
                160, 55,
                115, 95,
                130, 150,
                80, 120,
                30, 150,
                45, 95,
                0, 55,
                60, 50,
            ]);

            obj.starButton4.buttonMode = true;
            obj.starButton4.interactive = true;

            obj.starButton4.on('pointerdown', function() {
                onClick(obj.starButton4);
            });
            obj.starButton4.on('pointerover', function() {
                onPointerOver(obj.starButton4);
            });
            obj.starButton4.on('pointerout', function() {
                onPointerOut(obj.starButton4);
            });
        };

        var setText = function() {
            var style = new PIXI.TextStyle({ fill: '#ffffff' });

            obj.text1 = new PIXI.Text('Standard', style);
            obj.text1.x = obj.starButton1.x + 25;
            obj.text1.y = obj.starButton1.y + 170;

            obj.text2 = new PIXI.Text('Hit Area', style);
            obj.text2.x = obj.starButton2.x + 35;
            obj.text2.y = obj.starButton2.y + 170;

            obj.text3 = new PIXI.Text('Mask', style);
            obj.text3.x = obj.starButton3.x + 10;
            obj.text3.y = obj.starButton3.y + 170;

            obj.text4 = new PIXI.Text('Mask + Hit Area', style);
            obj.text4.x = obj.starButton4.x - 10;
            obj.text4.y = obj.starButton4.y + 170;
        };

        var setStage = function() {
            obj.app.stage.addChild(
                obj.starButton2,
                obj.starButton1,
                obj.starButton3,
                obj.starButton4,
                obj.squareMask,
                obj.squareMask2,
                obj.text1,
                obj.text2,
                obj.text3,
                obj.text4,
            );
        };

        var onClick = function(object) {
            object.tint = 0x333333;
        };

        var onPointerOver = function(object) {
            object.tint = 0x666666;
        };

        var onPointerOut = function(object) {
            object.tint = 0xFFFFFF;
        };

        return {
            init: init
        }
    })();

    pixi.component.interactionCustomHitarea.init();
})();

🔗 PIXI.Polygon
https://pixijs.download/release/docs/PIXI.Polygon.html
Description
사용자 정의 좌표를 통해 모양을 정의하는 클래스.

Methods
  • clone () : PIXI.Polygon

    이 다각형의 복제본을 만든다.

  • contains (x, y) : boolean

    이 함수에 전달된 x 및 y 좌표가 이 다각형 내에 포함되어 있는지 확인한다.

이름 타입 설명
x number X 좌표
y number Y 좌표

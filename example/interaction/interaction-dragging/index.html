PIXI Interaction - Dragging
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.interactionDragging = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-interaction-dragging');

            if (!!obj.section) {
                setElements();
                setDude();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('interactionDragging'),
                backgroundColor: 0x1099bb
            });

            obj.texture = PIXI.Texture.from('/content/images/dude3.png');
            obj.texture.baseTexture.scaleMode = PIXI.SCALE_MODES.NEAREST;
        };

        var setDude = function() {
            for (var i = 0; i < 10; i++) {
                createDude(
                    Math.floor(Math.random() * obj.app.screen.width),
                    Math.floor(Math.random() * obj.app.screen.height),
                );
            }
        };

        var createDude = function(x, y) {
            var dude = new PIXI.Sprite(obj.texture);
            dude.interactive = true;
            dude.buttonMode = true;
            dude.anchor.set(0.5);
            dude.scale.set(0.5);
            dude
                .on('pointerdown', onDragStart)
                .on('pointerup', onDragEnd)
                .on('pointerupoutside', onDragEnd)
                .on('pointermove', onDragMove);
            dude.x = x;
            dude.y = y;
            obj.app.stage.addChild(dude);
        };

        var onDragStart = function(event) {
            this.data = event.data;
            this.alpha = 0.5;
            this.dragging = true;
        };

        var onDragEnd = function() {
            this.alpha = 1;
            this.dragging = false;
            this.data = null;
        };

        var onDragMove = function() {
            if (this.dragging) {
                const newPosition = this.data.getLocalPosition(this.parent);
                this.x = newPosition.x;
                this.y = newPosition.y;
            }
        };

        return {
            init: init
        }
    })();

    pixi.component.interactionDragging.init();
})();

PIXI Interaction - Custom Mouse Icon
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.interactionCustomMouseIcon = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-interaction-custom-mouse-icon');

            if (!!obj.section) {
                setElements();
                setButtonInteraction();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('interactionCustomMouseIcon'),
                backgroundColor: 0x1099bb
            });

            var defaultIcon = "url('/content/images/bunny.png'),auto";
            var hoverIcon = "url('/content/images/bunny_saturated.png'),auto";

            obj.app.renderer.plugins.interaction.cursorStyles.hover = hoverIcon;
            obj.app.renderer.plugins.interaction.cursorStyles.default = defaultIcon;

            obj.background = PIXI.Sprite.from('/content/images/bg_button.jpg');
            obj.background.width = obj.app.screen.width;
            obj.background.height = obj.app.screen.height;

            obj.app.stage.addChild(obj.background);

            // Button
            obj.textureButton = PIXI.Texture.from('/content/images/button.png');
            obj.textureButtonDown = PIXI.Texture.from('/content/images/button_down.png');
            obj.textureButtonOver = PIXI.Texture.from('/content/images/button_over.png');

            obj.buttons = [];
            obj.buttonPositions = [
                175, 75,
                655, 75,
                410, 325,
                150, 465,
                685, 445,
            ];
        };

        var setButtonInteraction = function() {
            for (var i = 0; i < 5; i++) {
                var button = new PIXI.Sprite(obj.textureButton);
                button.cursor = 'hover';

                button.anchor.set(0.5);
                button.x = obj.buttonPositions[i * 2];
                button.y = obj.buttonPositions[i * 2 + 1];
                button.interactive = true;

                button
                    .on('pointerdown', onButtonDown)
                    .on('pointerup', onButtonUp)
                    .on('pointerupoutside', onButtonUp)
                    .on('pointerover', onButtonOver)
                    .on('pointerout', onButtonOut);
                obj.app.stage.addChild(button);

                obj.buttons.push(button);
            }

            obj.buttons[0].scale.set(1.2);
            obj.buttons[2].rotation = Math.PI / 10;
            obj.buttons[3].scale.set(0.8);
            obj.buttons[4].scale.set(0.8, 1.2);
            obj.buttons[4].rotation = Math.PI;
        };

        var onButtonDown = function() {
            this.isdown = true;
            this.texture = obj.textureButtonDown;
            this.alpha = 1;
        };

        var onButtonUp = function() {
            this.isdown = false;
            if (this.isOver) {
                this.texture = obj.textureButtonOver;
            } else {
                this.texture = obj.textureButton;
            }
        };

        var onButtonOver = function() {
            this.isOver = true;
            if (this.isdown) {
                return;
            }
            this.texture = obj.textureButtonOver;
        };

        var onButtonOut = function() {
            this.isOver = false;
            if (this.isdown) {
                return;
            }
            this.texture = obj.textureButton;
        };

        return {
            init: init
        }
    })();

    pixi.component.interactionCustomMouseIcon.init();
})();

PIXI Interaction - Click
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.interactionClick = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-interaction-click');

            if (!!obj.section) {
                setElements();
                bindEvents();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('interactionClick'),
                backgroundColor: 0x1099bb
            });
            PIXI.settings.SCALE_MODE = PIXI.SCALE_MODES.NEAREST;

            obj.sprite = PIXI.Sprite.from('/content/images/dude1.png');
            obj.sprite.anchor.set(0.5);
            obj.sprite.x = obj.app.screen.width / 2;
            obj.sprite.y = obj.app.screen.height / 2;
            obj.sprite.interactive = true;
            obj.sprite.buttonMode = true;
            obj.app.stage.addChild(obj.sprite);
        };

        var bindEvents = function() {
            obj.sprite.on('pointerdown', onClickSprite);
        };

        var onClickSprite = function() {
            obj.sprite.scale.x *= 1.25;
            obj.sprite.scale.y *= 1.25;
        };


        return {
            init: init
        }
    })();

    pixi.component.interactionClick.init();
})();

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.simplePlane = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-simple-plane');

            if (!!obj.section) {
                setElements();
                setLoader();
                bindEvents();
            }
        };

        var setElements = function() {
            obj.simplePlaneApp = new PIXI.Application({
                view: document.getElementById('simplePlane'),
                width: 800,
                height: 600,
                backgroundColor: 0x1099bb,
                resolution: window.devicePixelRatio || 1
            });
        };

        var bindEvents = function() {

        };

        var buildSimplePlane = function() {
            var texture = obj.simplePlaneApp.loader.resources.bg_grass.texture;

            var verticesX = 10;
            var verticesY = 10;
            var plane = new PIXI.SimplePlane(texture, verticesX, verticesY);

            plane.x = 100;
            plane.y = 100;

            obj.simplePlaneApp.stage.addChild(plane);

            var buffer = plane.geometry.getBuffer('aVertexPosition');

            obj.simplePlaneApp.ticker.add(function() {
                for (var i = 0; i < buffer.data.length; i++) {
                    buffer.data[i] += (Math.random() - 0.5);
                }
                buffer.update();
            });
        };

        var setLoader = function() {
            obj.simplePlaneApp.loader
            .add('bg_grass', '/content/images/bg_grass.jpg')
            .load(buildSimplePlane);
        };

        return {
            init: init
        }
    })();

    pixi.component.simplePlane.init();
})();

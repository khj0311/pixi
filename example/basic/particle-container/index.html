PIXI.Sprite
Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.particleContainer = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-particle-container');

            if (!!obj.section) {
                setElements();
                setMaggots();
                setTicker();
                bindEvents();
            }
        };

        var setElements = function() {
            obj.particleApp = new PIXI.Application({
                view: document.getElementById('particleContainer'),
                width: 800,
                height: 800
            });

            obj.sprites = new PIXI.ParticleContainer(100, {
                scale: true,
                position: true,
                rotation: true,
                uvs: true,
                alpha: true
            });

            obj.particleApp.stage.addChild(obj.sprites);

            obj.maggots = [];
            obj.maggotBoundsPadding = 100;
            obj.maggotBounds = new PIXI.Rectangle(
                -obj.maggotBoundsPadding,
                -obj.maggotBoundsPadding,
                obj.particleApp.screen.width + obj.maggotBoundsPadding * 2,
                obj.particleApp.screen.height + obj.maggotBoundsPadding * 2,
            );

            obj.tick = 0;

            // Sprite
            obj.addSprite = obj.section.querySelector('#addSprite');
            obj.removeSprite = obj.section.querySelector('#removeSprite');
            obj.removeSomeSprite = obj.section.querySelector('#removeSomeSprite');
        };

        var setMaggots = function() {
            for (var i = 0; i < 10; i++) {
                createSprite();
            }
        };

        var setTicker = function() {
            obj.particleApp.ticker.add(tickerAnimation);
        };

        var tickerAnimation = function() {
            for (var i = 0; i < obj.maggots.length; i++) {
                var maggot = obj.maggots[i];
                maggot.scale.y = 0.95 + Math.sin(obj.tick + maggot.offset) * 0.05;
                maggot.direction += maggot.turningSpeed * 0.01;
                maggot.x += Math.sin(maggot.direction) * (maggot.speed * maggot.scale.y);
                maggot.y += Math.cos(maggot.direction) * (maggot.speed * maggot.scale.y);
                maggot.rotation = -maggot.direction + Math.PI;

                if (maggot.x < obj.maggotBounds.x) {
                    maggot.x += obj.maggotBounds.width;
                } else if (maggot.x > obj.maggotBounds.x + obj.maggotBounds.width) {
                    maggot.x -= obj.maggotBounds.width;
                }

                if (maggot.y < obj.maggotBounds.y) {
                    maggot.y += obj.maggotBounds.height;
                } else if (maggot.y > obj.maggotBounds.y + obj.maggotBounds.height) {
                    maggot.y -= obj.maggotBounds.height;
                }
            }

            obj.tick += 0.1;
        };

        var bindEvents = function() {
            if (obj.addSprite) obj.addSprite.addEventListener('click', onClickAddSprite);
            if (obj.removeSprite) obj.removeSprite.addEventListener('click', onClickRemoveSprite);
        };

        var createSprite = function() {
            var maggot = PIXI.Sprite.from('/content/images/maggot_tiny.png');
            maggot.anchor.set(0.5);
            maggot.scale.set(0.8 + Math.random() * 0.3);
            maggot.x = Math.random() * obj.particleApp.screen.width;
            maggot.y = Math.random() * obj.particleApp.screen.height;
            maggot.tint = Math.random() * 0x808080;
            maggot.direction = Math.random() * Math.PI * 2;
            maggot.turningSpeed = Math.random() - 0.8;
            maggot.speed = (2 + Math.random() * 2) * 0.2;
            maggot.offset = Math.random() * 100;
            obj.maggots.push(maggot);
            obj.sprites.addChild(maggot);
        };

        var onClickAddSprite = function() {
            createSprite();
        };

        var onClickRemoveSprite = function() {
            var child = obj.sprites.children;
            var leng = obj.sprites.children.length - 1;
            obj.maggots.pop();
            obj.sprites.removeChild(child[leng]);
        };

        return {
            init: init
        }
    })();

    pixi.component.particleContainer.init();
})();

🔗 PIXI.ParticleContainer
https://pixijs.download/release/docs/PIXI.ParticleContainer.html
Description
ParticleContainer 클래스는 오로지 속도만을 위해 만들어진 PIXI.Container의 정말 빠른 버전이므로 많은 Sprite나 Particle이 필요할 때 사용한다.
ParticleContainer의 단점은 대부분의 고급 기능이 작동하지 않는다는 것이다. ParticleContainer는 기본 개체 변환(위치, 크기, 회전) 및 색조와 같은 일부 고급 기능(v4.5.6 기준)을 구현한다.
Masking, Children, Filters 등과 같은 기타 고급 기능은 이 batch의 Sprite에서는 작동하지 않는다.

Example

var container = new ParticleContainer();

for (var i = 0; i < 100; ++i) {
    var sprite = PIXI.Sprite.from("myImage.png");
    container.addChild(sprite);
}

Properties
new PIXI.ParticleContainer (maxSize, properties, batchSize, autoResize)
이름 타입 기본값 설명
maxSize number 1500 Container에서 Rendering할 수 있는 최대 Particle 수. 할당된 버퍼의 크기에 영향을 준다.
properties object GPU에 업로드하여 적용해야하는 자식의 속성
properties.vertices boolean false true일 경우 정점이 업로드되고 적용된다. Sprite의 scale / anchor / trim / frame / orig가 동적이면 true로 설정한다.
properties.position boolean true true일 경우 position이 업로드되어 적용된다.
properties.rotation boolean false true일 경우 rotation이 업로드되어 적용된다.
properties.uvs boolean false true일 경우 uv가 업로드되어 적용된다.
properties.tint boolean false true일 경우 alpha와 tint가 업로드되어 적용된다.
batchSize number 16384 배치당 Particle 수. maxSize보다 작으면 maxSize가 대신 사용된다.
autoResize boolean false true일 경우 max Size보다 더 많은 Particle이 존재할 경우에 대비하여 Container에 더 많은 batch가 할당된다. 

var app = new PIXI.Application({
    view: document.getElementById('particleContainer'),
    width: 800,
    height: 800
});

var sprites = new PIXI.ParticleContainer(100, {
    scale: true,
    position: true,
    rotation: true,
    uvs: true,
    alpha: true
});

app.stage.addChild(sprites);

Methods
  • destroy (options)

    컨테이너를 파괴한다.

이름 타입 기본값 설명
options object | boolean 옵션 매개변수. true일 경우 모든 옵션이 해당 값으로 설정된 것처럼 작동한다.
options.children boolean false true일 경우 모든 자식의 destroy 메소드도 호출된다.
options.texture boolean false options.children이 true일 경우 자식 Sprite에만 사용된다. 자식 Sprite의 Texture가 파괴된다.
options.baseTexture boolean false options.children이 true일 경우 자식 Sprite에만 사용된다. 자식 Sprite의 기본 Texture가 파괴된다.

    전달된 속성 개체를 기반으로 private 속성 배열을 동적/정적으로 설정한다.

properties object 업로드할 속성

🔗 PIXI.Rectangle
https://pixijs.download/release/docs/PIXI.Rectangle.html
Description
Rectangle 개체는 왼쪽 위 모서리 지점(x, y)과 너비 및 높이로 표시되는 위치로 정의되는 영역이다.

Properties
new PIXI.Rectangle (x, y, width, height)
이름 타입 기본값 설명
x number 0 직사각형의 왼쪽 위 모서리의 X 좌표
y number 0 직사각형의 왼쪽 위 모서리의 Y 좌표
width number 0 직사각형의 전체 너비
height number 0 직사각형의 전체 높이
Methods
  • destroy (options)

    모서리가 그리드에 놓이도록 직사각형을 확대한다.

이름 타입 기본값 설명
resolution number 1 해상도
eps number 0.001 정밀도
  • clone ()

    이 직사각형의 복제본을 만든다.

  • contains (x, y)

    주어진 x 및 y 좌표가 이 직사각형 내에 포함되어 있는지 확인한다.

이름 타입 설명
x number 테스트할 점의 X 좌표
y number 테스트할 점의 Y 좌표
  • containsRect (other)

    이 직사각형 개체 내에 다른 직사각형이 포함되어 있는지 여부를 결정한다.

    동일한 공간을 차지하는 직사각형은 서로를 포함하는 것으로 간주한다.

    영역이 없는 직사각형(너비 또는 높이가 0임)은 다른 영역이 없는 직사각형도 포함하여 아무 것도 포함할 수 없다.

  • equals (other)

    다른 직사각형을 허용하고 주어진 직사각형이 이 직사각형과 같으면 true를 반환한다.

이름 타입 설명
x Rectangle 포함되어 있는지 확인 할 Rectangle 객체
  • copyFrom (rectangle)

    이 직사각형에 다른 직사각형을 복사한다.

  • copyTo (rectangle)

    이 직사각형을 다른 직사각형에 복사한다.

  • fit (rectangle)

    전달된 직사각형 주위에 이 직사각형을 맞춘다.

이름 타입 설명
rectangle Rectangle 복사하거나 맞출 Rectangle 객체

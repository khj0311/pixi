Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.container = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-container');

            if (!!obj.section) {
                setElements();
                setSprite();
                setContainer();
                bindEvents();
            }
        };

        var setElements = function() {
            obj.containerApp = new PIXI.Application({
                view: document.getElementById('container'),
                width: 800,
                height: 800,
                backgroundColor: 0xa0f584,
                resolution: window.devicePixelRatio || 1
            });

            obj.container = new PIXI.Container();
            obj.containerApp.stage.addChild(obj.container);
            obj.texture = PIXI.Texture.from('/content/images/bunny.png');

            // Sprite
            obj.addSprite = obj.section.querySelector('#addSprite');
            obj.removeSprite = obj.section.querySelector('#removeSprite');
            obj.removeSomeSprite = obj.section.querySelector('#removeSomeSprite');

            // Ticker
            obj.addTicker = obj.section.querySelector('#addTicker');
            obj.removeTicker = obj.section.querySelector('#removeTicker');
            obj.startTicker = obj.section.querySelector('#startTicker');
            obj.stopTicker = obj.section.querySelector('#stopTicker');
        };

        var setSprite = function() {
            for (var i = 0; i < 25; i++) {
                var bunny = new PIXI.Sprite(obj.texture);
                bunny.anchor.set(0.5);
                bunny.scale.set(2);
                bunny.x = (i % 5) * 100;
                bunny.y = Math.floor(i / 5) * 100;
                obj.container.addChild(bunny);
            }
        };

        var setContainer = function() {
            obj.container.x = obj.containerApp.screen.width / 2;
            obj.container.y = obj.containerApp.screen.height / 2;

            obj.container.pivot.x = obj.container.width / 2;
            obj.container.pivot.y = obj.container.height / 2;
        };

        var bindEvents = function() {
            obj.container.on('childAdded', function(child, cont, index) {
                console.log(child);
                console.log(cont);
                console.log(index);
            });

            if (obj.addSprite) obj.addSprite.addEventListener('click', onClickAddSprite);
            if (obj.removeSprite) obj.removeSprite.addEventListener('click', onClickRemoveSprite);
            if (obj.removeSomeSprite) obj.removeSomeSprite.addEventListener('click', onClickRemoveSomeSprite);

            if (obj.addTicker) obj.addTicker.addEventListener('click', onClickAddTicker);
            if (obj.removeTicker) obj.removeTicker.addEventListener('click', onClickRemoveTicker);
            if (obj.startTicker) obj.startTicker.addEventListener('click', onClickStartTicker);
            if (obj.stopTicker) obj.stopTicker.addEventListener('click', onClickStopTicker);
        };

        var onClickAddSprite = function() {
            var leng = obj.container.children.length;
            var bunny = new PIXI.Sprite(obj.texture);
            bunny.anchor.set(0.5);
            bunny.scale.set(2);
            bunny.x = (leng % 5) * 100;
            bunny.y = Math.floor(leng / 5) * 100;
            obj.container.addChild(bunny);
        };

        var onClickRemoveSprite = function() {
            var child = obj.container.children;
            var leng = obj.container.children.length - 1;
            obj.container.removeChild(child[leng]);
        };

        var onClickRemoveSomeSprite = function() {
            var startIndex = document.querySelector('#startIndex').value;
            var endIndex = document.querySelector('#endIndex').value;
            startIndex = !!startIndex ? startIndex : 0;
            endIndex = !!endIndex ? endIndex : obj.container.children.length;

            obj.container.removeChildren(startIndex, endIndex);
        };

        var tickerAnimation = function(delta) {
            obj.container.rotation -= 0.01 * delta;
        };

        var onClickAddTicker = function() {
            obj.containerApp.ticker.add(tickerAnimation);
        };

        var onClickRemoveTicker = function() {
            obj.containerApp.ticker.remove(tickerAnimation);
        };

        var onClickStartTicker = function() {
            obj.containerApp.ticker.start();
        };

        var onClickStopTicker = function() {
            obj.containerApp.ticker.stop();
        };

        return {
            init: init
        }
    })();

    pixi.component.container.init();
})();
🔗 PIXI.DisplayObject
https://pixijs.download/release/docs/PIXI.DisplayObject.html
Description
화면에 Rendering되는 모든 개체의 기본 Class.
이것은 추상 Class이며 자체적으로 사용할 수 없다.

PixiJS에서 구현된 디스플레이 객체
Display Object 설명
PIXI.Container DisplayObject를 추가할 수 있는 컨테이너 객체
PIXI.Graphics Canvas API와 유사한 모양 그리기 표시 객체
PIXI.Sprite 텍스처(이미지)를 그림
PIXI.Text 내부적으로 Canvas API를 사용하여 텍스트를 그린다.
PIXI.TilingSprite 타일 방식으로 Texture/Image를 그린다.
PIXI.AnimatedSprite 여러 이미지의 애니메이션을 그린다.
PIXI.Mesh 사용자 지정 데이터로 Mesh를 그리기 위한 하위 수준 API 제공

Transforms
디스플레이 객체의 Transform은 로컬 좌표공간에서 부모 좌표공간으로의 투영을 기술한다.
다음 속성은 Transform에서 얻는다.
Property 설명
pivot rotation, scaling 및 skew 아래 변경 불가하다. pivot의 부모 공간에 대한 투영은 다른 세 가지 변환과 관계없이 위치와 동일하다. 즉, rotation, scaling 및 skew의 중심이다.
position 부모 로컬공간의 pivot 위치이다. pivot의 기본값은 origin(0,0)이다. 디스플레이 객체의 좌측 상단 모서리가 로컬공간에 있는 경우, 그 위치는 부모의 로컬공간에 있는 좌측 상단 모서리가 된다.
scale 디스플레이 개체의 투영이 늘어나거나 압축된다. 스케일 팩터는 로컬 좌표 축을 따라 있다. 즉, 디스플레이 표시 객체는 회전하거나 치우치기 전에 축척이 조정된다. 스케일링의 중심은 pivot이다.
rotation 디스플레이 표시 객체의 투영이 이 각도(radian 단위)로 회전한다.
angle 각도 단위로 된 회전 별칭.
skew 직사각형 디스플레이 표시 객체를 평행사변형으로 변형하는 데 사용할 수 있다.
x x 좌표
y y 좌표
width Container에 구현된다. width 속성은 요청된 너비를 로컬공간 상자 너비로 나누어 scale.x를 계산한다. scale.x에 대한 간접적인 추상화이며 사용자 정의 너비에 대한 개념은 없다.
height Container에 구현된다. height 속성은 요청된 높이를 로컬공간 상자 높이로 나누어 scale.y를 계산한다. scale.y에 대한 간접적인 추상화이며 사용자 정의 높이에 대한 개념은 없다.
🔗 PIXI.Application
https://pixijs.download/release/docs/PIXI.Application.html
Description
새로운 PIXI 애플리케이션을 생성하기 위한 편의 클래스이며, Renderer, Ticker 및 루트 컨테이너를 자동으로 생성한다.

Example

// Application 생성
const app = new PIXI.Application();

// DOM에 view 추가
document.body.appendChild(app.view);

// 예시) 표시할 스프라이트 객체 추가
app.stage.addChild(PIXI.Sprite.from('something.png'));
                    

var app = new PIXI.Application({
    view: document.getElementById('container'),
    width: 800,
    height: 800,
    backgroundColor: 0xa0f584,
    resolution: window.devicePixelRatio || 1
});

Properties

이름 타입 기본값 설명
options object 선택적 Renderer 매개변수

🔗 PIXI.Container
https://pixijs.download/release/docs/PIXI.Container.html
Description
Container는 child를 포함하는 범용 표시 객체이며, Masking 및 Filtering과 같은 고급 Rendering 기능에 대한 기본 기능을 제공한다.
또한, Graphics 및 Sprite를 비롯한 다른 객체의 컨테이너 역할을 하는 모든 표시 객체의 기본 Class 이다.

Example

        var app = new PIXI.Application({
            view: document.getElementById('container'),
            width: 800,
            height: 800,
            backgroundColor: 0xa0f584,
            resolution: window.devicePixelRatio || 1
        });
        
        var container = new PIXI.Container();
        
        app.stage.addChild(container);
        
        var texture = PIXI.Texture.from('/content/images/bunny.png');
        
        for (var i = 0; i < 25; i++) {
            var bunny = new PIXI.Sprite(texture);
            bunny.anchor.set(0.5);
            bunny.scale.set(2);
            bunny.x = (i % 5) * 100;
            bunny.y = Math.floor(i / 5) * 100;
            container.addChild(bunny);
        }

Methods
  • addChild (...children)

    컨테이너에 하나 이상의 자식을 추가한다.

    다음과 같이 여러 항목을 추가할 수 있다. myContainer.addChild(thingOne, thingTwo, thingThree)

  • removeChild (...children)

    컨테이너에 하나 이상의 자식을 제거한다.

이름 타입 설명
children PIXI.DisplayObject 컨테이너에 추가할 DisplayObject(들)
  • removeChildren (beginIndex, endIndex)

    시작 및 끝 인덱스 내에 있는 이 컨테이너에서 모든 자식을 제거한다.

    (중간의 일부를 제거할 수는 없다.)

이름 타입 기본값 설명
beginIndex number 0 시작 인덱스
endIndex number this.children.length 끝 인덱스, 기본 값은 컨테이너의 크기

Events

        container.on('childAdded', function(child, cont, index) {
            console.log(child);

이름 타입 설명

🔗 PIXI.Texture
https://pixijs.download/release/docs/PIXI.Texture.html
Description
Texture는 이미지 또는 이미지의 일부를 나타내는 정보를 저장한다.
Display List에 직접 추가할 수 없으며 Sprite의 Texture로 사용해야 한다.
Texture에 프레임이 제공되지 않으면 전체 이미지가 사용된다.
　
이미지에서 직접 Texture를 만든 다음 다음과 같이 여러 번 재사용할 수 있다.

            var texture = PIXI.Texture.from('/content/images/bunny.png');
            var bunny1 = new PIXI.Sprite(texture);
            var bunny2 = new PIXI.Sprite(texture);
텍스처 프레임을 생성자에 전달하지 않으면 noFrame 모드가 활성화된다.
baseTexture 이벤트를 구독하고 baseTexture와 동시에 자동으로 크기가 조정된다.
　
로드 여부와 관계없이 SVG로 만든 텍스처는 파일 처리가 완료되기 전에 사용할 수 없다.
PIXI.Sprite의 _textureID 속성을 확인하여 이를 확인할 수 있다.

            var texture = PIXI.Texture.from('assets/image.svg');
            var sprite1 = new PIXI.Sprite(texture);
            //sprite1._textureID

Properties
new PIXI.Texture (baseTexture, frame, orig, trim, rotate, anchor)
이름 타입 설명
baseTexture PIXI.BaseTexture Texture를 생성 할 기본 Texture source
frame PIXI.Rectangle 표시할 Texture의 사각 프레임
orig PIXI.Rectangle 기본 Texture 영역
trim PIXI.Rectangle 기본 Texture의 트리밍 된 사각형
rotate number Texture packer가 Texture를 회전할 각도를 표시
anchor PIXI.IPointData Sprite 배치/회전에 사용되는 기본 앵커 포인트

Methods

    제공한 소스를 기반으로 새 Texture를 생성하는 도우미 함수.

    source는 프레임 ID, 이미지 URL, 비디오 URL, 캔버스 요소, 비디오 요소, 기본 텍스처일 수 있다.

이름 타입 설명

options object PIXI.BaseTexture
strict number strict mode 적용
  • clone ()

    해당 Texture와 같은 역할을 하는 새로운 Texture 개체를 만든다.

  • destroy ()

    해당 Texture를 파괴한다.

  • update ()

    GPU에서 해당 Texture를 업데이트한다.

🔗 PIXI.Sprite
https://pixijs.download/release/docs/PIXI.Sprite.html
Description
Sprite 개체는 화면에 렌더링되는 모든 Texture 개체의 기반이다.

        PIXI.Loader.shared.add("assets/spritesheet.json").load(setup);
        
        function setup() {
            var sheet = PIXI.Loader.shared.resources["assets/spritesheet.json"].spritesheet;
            var sprite = new PIXI.Sprite(sheet.textures["image.png"]);
            ...
        }

🔗 PIXI.Ticker
https://pixijs.download/release/docs/PIXI.Ticker.html
Description
다른 개체가 수신하는 업데이트 루프를 실행하는 Ticker 클래스.
이 클래스는 요청된 애니메이션 프레임에서 실행하기 위한 리스너를 중심으로 구성된다.
Ticker가 시작되고 Emitter에 리스너가 있는 경우 애니메이션 프레임은 필요할 때만 요청된다.

        app.ticker.add(function(delta) {
        container.rotation -= 0.01 * delta;
        });

Methods
  • add (fn, context, priority)

    Tick 이벤트에 대한 Handler를 추가한다.

    제거되거나 Ticker가 중지되지 않는 한 계속해서 호출한다.

  • addOnce (fn, context, priority)

    한 번만 실행되는 Tick 이벤트에 대한 Handler를 추가한다.

  • remove (fn, context)

    함수 및 context 매개변수와 일치하는 모든 Handler를 제거한다.

    제거한 후 Handler가 남지 않으면 애니메이션 프레임이 취소된다.

이름 타입 기본값 설명
fn TickerCallback 한 번의 업데이트에 추가/삭제 될 Ticker Listener 기능
context T Listener 컨텍스트
priority number PIXI.UPDATE_PRIORITY.NORMAL 실행 순서 우선순위 

        var tickerAnim = function(delta) {
        container.rotation -= 0.01 * delta;
        };
        
        var addTickerFunc = function() {
        app.ticker.add(tickerAnim);
        };
        
        var removeTickerFunc = function() {
        app.ticker.remove(tickerAnim);
        };
        
        var addTicker = document.querySelector('#addTicker');
        if (addTicker) addTicker.addEventListener('click', addTickerFunc);
        
        var removeTicker = document.querySelector('#removeTicker');
        if (removeTicker) removeTicker.addEventListener('click', removeTickerFunc);
  • destroy ()

    이 메서드를 호출하면 내부 이벤트에 대한 모든 참조가 제거된다.

    Ticker를 파괴하고 이후에는 사용하지 말아야 한다.

  • start ()

    Ticker를 시작한다.

    Ticker에 리스너가 있는 경우 이 시점(처음 시작 또는 정지되어있던 프레임)에서 새 애니메이션 프레임이 요청된다.

  • stop ()

    Ticker를 중지한다.

    Ticker가 애니메이션 프레임을 요청한 경우 이 시점에서 취소된다.


        var startTicker = document.querySelector('#startTicker');
        if (startTicker) {
        startTicker.addEventListener('click', function() {
            app.ticker.start();
        });
        }
        
        var stopTicker = document.querySelector('#stopTicker');
        if (stopTicker) {
        stopTicker.addEventListener('click', function() {
            app.ticker.stop();
        });
        }

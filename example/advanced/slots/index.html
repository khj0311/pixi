Example Code

;(function() {
    'use strict';

    window.pixi = window.pixi || {};
    window.pixi.component = window.pixi.component || {};

    pixi.component.slots = (function() {
        var obj = {};

        var init = function() {
            obj.section = document.querySelector('.pixi-slots');

            if (!!obj.section) {
                setElements();
                setLoader();
                setTicker();
                bindEvents();
            }
        };

        var setElements = function() {
            obj.app = new PIXI.Application({
                view: document.getElementById('slots'),
                backgroundColor: 0x1099bb
            });

            obj.reelWidth = 160;
            obj.symbolSize = 150;
            obj.running = false;
            obj.tweening = [];
            obj.reels = [];
        };

        var bindEvents = function() {

        };

        var setLoader = function() {
            obj.app.loader
                .add('/content/images/dude1.png', '/content/images/dude1.png')
                .add('/content/images/dude2.png', '/content/images/dude2.png')
                .add('/content/images/dude3.png', '/content/images/dude3.png')
                .add('/content/images/dude4.png', '/content/images/dude4.png')
                .load(onAssetsLoaded);
        };

        var setTicker = function() {
            obj.app.ticker.add(function() {
                var now = Date.now();
                var remove = [];
                for (var i = 0; i < obj.tweening.length; i++) {
                    var t = obj.tweening[i];
                    var phase = Math.min(1, (now - t.start) / t.time);

                    t.object[t.property] = lerp(t.propertyBeginValue, t.target, t.easing(phase));
                    if (t.change) t.change(t);
                    if (phase === 1) {
                        t.object[t.property] = t.target;
                        if (t.complete) t.complete(t);
                        remove.push(t);
                    }
                }

                for (var i = 0; i < remove.length; i++) {
                    obj.tweening.splice(obj.tweening.indexOf(remove[i]), 1);
                }
            });
        };

        var startPlay = function() {
            if (obj.running) return;
            obj.running = true;

            for (var i = 0; i < obj.reels.length; i++) {
                var r = obj.reels[i];
                var extra = Math.floor(Math.random() * 3);
                var target = r.position + 10 + i * 5 + extra;
                var time = 2500 + i * 600 + extra * 600;
                tweenTo(r, 'position', target, time, backout(0.5), null, i === obj.reels.length - 1 ? reelsComplete : null);
            }
        };
        
        var reelsComplete = function() {
            obj.running = false;
        };

        var onAssetsLoaded = function() {
            var slotTextures = [
                PIXI.Texture.from('/content/images/dude1.png'),
                PIXI.Texture.from('/content/images/dude2.png'),
                PIXI.Texture.from('/content/images/dude3.png'),
                PIXI.Texture.from('/content/images/dude4.png'),
            ];

            var reelContainer = new PIXI.Container();
            for (var i = 0; i < 5; i++) {
                var rc = new PIXI.Container();
                rc.x = i * obj.reelWidth;
                reelContainer.addChild(rc);

                var reel = {
                    container: rc,
                    symbols: [],
                    position: 0,
                    previousPosition: 0,
                    blur: new PIXI.filters.BlurFilter
                };

                reel.blur.blurX = 0;
                reel.blur.blurY = 0;
                rc.filters = [reel.blur];

                for (var j = 0; j < 4; j++) {
                    var symbol = new PIXI.Sprite(slotTextures[Math.floor(Math.random() * slotTextures.length)]);
                    symbol.y = j * obj.symbolSize;
                    symbol.scale.x = symbol.scale.y = Math.min((obj.symbolSize / symbol.width), obj.symbolSize / symbol.height);
                    symbol.x = Math.round((obj.symbolSize - symbol.width) / 2);
                    reel.symbols.push(symbol);
                    rc.addChild(symbol);
                }
                obj.reels.push(reel);
            }
            obj.app.stage.addChild(reelContainer);

            var margin = (obj.app.screen.height - (obj.symbolSize * 3)) / 2;
            reelContainer.y = margin;
            reelContainer.x = Math.round(obj.app.screen.width - (obj.reelWidth * 5));
            var top = new PIXI.Graphics();
            top.beginFill(0, 1);
            top.drawRect(0, 0, obj.app.screen.width, margin);
            var bottom = new PIXI.Graphics();
            bottom.beginFill(0, 1);
            bottom.drawRect(0, (obj.symbolSize * 3) + margin, obj.app.screen.width, margin);

            var style = new PIXI.TextStyle({
                fontFamily: 'Arial',
                fontSize: 36,
                fontStyle: 'italic',
                fontWeight: 'bold',
                fill: ['#76e776', '#eceda2'],
                stroke: '#4a1850',
                strokeThickness: 5,
                dropShadow: true,
                dropShadowColor: '#000000',
                dropShadowBlur: 4,
                dropShadowAngle: Math.PI / 6,
                dropShadowDistance: 6,
                wordWrap: true,
                wordWrapWidth: 440,
            });

            var playText = new PIXI.Text('Spin the wheels!', style);
            playText.x = Math.round((bottom.width - playText.width) / 2);
            playText.y = obj.app.screen.height - margin + Math.round((margin - playText.height) / 2);
            bottom.addChild(playText);

            var headerText = new PIXI.Text('PIXI CRAYON SLOTS!', style);
            headerText.x = Math.round((top.width - headerText.width) / 2);
            headerText.y = Math.round((margin - headerText.height) / 2);
            top.addChild(headerText);
        
            obj.app.stage.addChild(top);
            obj.app.stage.addChild(bottom);
        
            bottom.interactive = true;
            bottom.buttonMode = true;
            bottom.addListener('pointerdown', () => {
                startPlay();
            });

            obj.app.ticker.add((delta) => {
                for (var i = 0; i < obj.reels.length; i++) {
                    var r = obj.reels[i];
                    r.blur.blurY = (r.position - r.previousPosition) * 8;
                    r.previousPosition = r.position;

                    for (var j = 0; j < r.symbols.length; j++) {
                        var s = r.symbols[j];
                        var prevy = s.y;
                        s.y = ((r.position + j) % r.symbols.length) * obj.symbolSize - obj.symbolSize;
                        if (s.y < 0 && prevy > obj.symbolSize) {
                            s.texture = slotTextures[Math.floor(Math.random() * slotTextures.length)];
                            s.scale.x = s.scale.y = Math.min(obj.symbolSize / s.texture.width, obj.symbolSize / s.texture.height);
                            s.x = Math.round((obj.symbolSize - s.width) / 2);
                        }
                    }
                }
            });
        };

        var tweenTo = function(object, property, target, time, easing, onchange, oncomplete) {
            var tween = {
                object,
                property,
                propertyBeginValue: object[property],
                target,
                easing,
                time,
                change: onchange,
                complete: oncomplete,
                start: Date.now(),
            };

            obj.tweening.push(tween);
            return tween;
        };

        var lerp = function(a1, a2, t) {
            return a1 * (1 - t) + a2 * t;
        };

        var backout = function(amount) {
            return (t) => (--t * t * ((amount + 1) * t + amount) + 1);
        };

        return {
            init: init
        }
    })();

    pixi.component.slots.init();
})();

🔗 PIXI.Graphics
https://pixijs.download/release/docs/PIXI.Graphics.html
Description
Graphics 클래스는 주로 선, 원 및 직사각형과 같은 기본 모양을 디스플레이에 렌더링하고 색상을 지정하고 채우는 데 사용된다.
그러나 Graphics 개체를 사용하여 마스크 또는 복잡한 hitArea로 사용할 기본 요소 목록을 작성할 수도 있다.

Regacy 명명 규칙으로 인해 이 클래스의 일부 함수 동작이 혼동될 수 있다.
drawRect(), drawPolygon() 등에 대한 각 호출은 실제로 나중에 렌더링 또는 적중 테스트에 사용하기 위해 Geometry 클래스의 GraphicsGeometry 객체에 해당 Primitive를 저장한다.
마찬가지로 clear() 함수는 화면을 변경하지 않고 단순히 기본 요소 목록을 재설정하므로 기존 Graphics 객체의 내용을 다시 작성하려는 경우에 유용할 수 있다.

GraphicsGeometry 목록이 작성되면 이를 새 Geometry 개체의 생성자에 전달하여 최적화로 다른 Geometry 개체에서 재사용할 수 있다.
이 기능 때문에 각 GraphicsGeometry를 적절하게 역참조하고 메모리 누수를 방지하기 위해 Geometry 개체에 대해 destroy()를 호출하는 것이 중요하다.

var top = new PIXI.Graphics();
top.beginFill(0, 1);
top.drawRect(0, 0, obj.app.screen.width, margin);
var bottom = new PIXI.Graphics();
bottom.beginFill(0, 1);
bottom.drawRect(0, (obj.symbolSize * 3) + margin, obj.app.screen.width, margin);

Properties
new PIXI.Graphics (geometry)
이름 타입 기본값 설명
geometry PIXI.Container null 사용할 geometry를 생략하면 새 GraphicsGeometry 인스턴스가 생성된다.

Members
  • blendMode : number

    그래픽 모양에 적용할 혼합 모드. 블렌드 모드를 재설정하려면 PIXI.BLEND_MODES.NORMAL 값을 적용해야한다.

    GraphicsGeometry 목록의 각 프리미티브는 순차적으로 렌더링되므로 PIXI.BLEND_MODES.ADD 및 PIXI.BLEND_MODES.MULTIPLY와 같은 모드가 프리미티브별로 적용된다.

  • cacheAsBitmap : boolean

    cacheAsBitmap이 true로 설정되면 그래픽 객체는 마치 스프라이트인 것처럼 렌더링된다.

    이것은 그래픽 요소가 자주 변경되지 않는 경우에 유용한데, 텍스처 메모리를 차지하는 대신 개체의 렌더링 속도를 높일 수 있기 때문이다.

    또한 그래픽 개체가 Canvas를 사용하여 렌더링되기 때문에 앤티앨리어싱이 필요한 경우에도 유용하다.

    그래픽 요소를 지속적으로 다시 그리는 경우에는 권장하지 않는다.

  • currentPath : PIXI.Polygon

    현재 경로

  • fill : PIXI.FillStyle

    현재 채우기 스타일

  • geometry : PIXI.GraphicsGeometry

    정점 위치, 면 인덱스, 법선, 색상, UV 및 버퍼 내의 사용자 정의 속성을 포함하여 이 모든 데이터를 GPU에 전달하는 비용을 줄인다.

    여러 Mesh 또는 Graphics 개체 간에 공유할 수 있다.

  • line : PIXI.LineStyle

    현재 선 스타일

  • pluginName : string

    일괄 처리를 위한 렌더링 플러그인

  • shader : PIXI.Shader

    Geometry를 처리하고 GPU에서 실행되는 꼭짓점 및 조각 Shader를 나타낸다. 여러 Graphics 개체 간에 공유할 수 있다.

  • tint : number

    각 Graphic 모양에 적용되는 틴트. 16진수 값을 사용하며 값을 0xFFFF로 설정하면 틴트의 영향이 없어진다.


Methods
  • arc (cx, cy, radius, startAngle, endAngle, anticlockwise)

    호/곡선을 생성한다. (원 또는 원의 일부를 생성하는 데 사용)

이름 타입 기본값 설명
cx number 원 중심의 x 좌표
cy number 원 중심의 y 좌표
radius number 원의 반경
startAngle number 시작 각도 (0은 호 원의 3시 위치에 있음)
endAngle number 종료 각도
anticlockwise boolean false 도면이 시계 반대 방향인지 시계 방향인지 여부를 지정한다. False는 기본값이며 시계 방향을 나타내고 true는 시계 반대 방향을 나타낸다.
  • arcTo (x1, y1, x2, y2, radius)

    Canvas의 두 접선 사이에 호/곡선을 만든다.

이름 타입 설명
x1 number 호의 첫 번째 접선점의 x 좌표
y1 number 호의 첫 번째 접선점의 y 좌표
x2 number 호 끝의 x 좌표
y2 number 호 끝의 y 좌표
radius number 호의 반경
  • beginFill (color, alpha)

    다른 Graphics 메서드(예: lineTo() 또는 drawCircle())에 대한 후속 호출이 그릴 때 사용하는 간단한 단색 채우기를 지정한다.

이름 타입 기본값 설명
color number 0 채우기의 색
alpha number 1 채우기의 알파값
  • clear ()

    이 Graphics 개체에 그려진 그래픽을 지우고 채우기 및 선 스타일 설정을 재설정한다.

  • clone ()

    이 값과 동일한 값으로 새 Graphics 개체를 만든다. 객체의 Geometry만 복제되고 변환(위치, 배율 등)이 복제되지 않는다.

  • closePath ()

    그리고있던 현재 경로를 닫는다. (새로운 경로를 그릴 경우)

  • destroy (options)

    Graphics 개체를 파괴한다.

이름 타입 기본값 설명
options object | boolean 옵션 매개변수. true 일 경우 모든 옵션이 해당 값으로 설정된 것처럼 작동
options.children boolean false true일 경우 모든 자식의 destroy 메소드를 호출
options.texture boolean false Sprite의 Texture도 파괴
options.baseTexture boolean false Sprite의 기본 Texture도 파괴
  • drawCircle (x, y, radius)

    원을 그린다.

이름 타입 설명
x number 원 중심의 X 좌표
y number 원 중심의 Y 좌표
radius number 원의 반경
  • drawEllipse (x, y, width, height)

    타원을 그린다.

이름 타입 설명
x number 타원 중심의 X 좌표
y number 타원 중심의 Y 좌표
width number 타원의 절반 너비
height number 타원의 절반 높이
  • drawPolygon (…path)

    주어진 경로를 사용하여 다각형을 그린다.

이름 타입 설명
path Array | Array | PIXI.Polygon Polygon을 구성하는 데 사용되는 경로 데이터
  • drawRect (x, y, width, height)

    직사각형 모양을 그린다.

이름 타입 설명
x number 직사각형의 왼쪽 상단의 X 좌표

🔗 PIXI.GraphicsGeometry
https://pixijs.download/release/docs/PIXI.GraphicsGeometry.html
Description
Graphics 클래스에는 선, 원 및 직사각형과 같은 기본 모양을 디스플레이에 그리고 색상을 지정하고 채우는 데 사용되는 메서드가 포함되어 있다.
GraphicsGeometry는 이어컷을 사용하여 다시 테셀레이션하는 데 비용이 많이 들기 때문에 형상을 지속적으로 업데이트하지 않도록 설계되어있다.

Methods
  • clear ()

    이 Graphics 개체에 그려진 그래픽을 지우고 채우기 및 선 스타일 설정을 재설정한다.

  • destroy ()

    GraphicsGeometry 개체를 파괴한다.

  • containsPoint (point)

    Point가 이 Geometry 내에 포함되어 있는지 여부를 확인한다.

이름 타입 설명
point IPointData 포함되어 있는지 확인할 Point.
